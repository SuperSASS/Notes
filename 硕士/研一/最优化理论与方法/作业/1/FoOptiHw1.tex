\begin{enumerate}

&\mathop{\Min}\limits_{\gb,\beta_0,\bm\xi} \quad
&& \frac{1}{2} \|\gb\|_2^2 + C \sum_{i=1}^n \xi_i \\
&\ST \quad && \xi_i \geq 0, \; i=1,\ldots n \\

\end{enumerate}
